                             monitor="val_accuracy", 
                             verbose=1, 
                             save_best_only=True, 
                             period=1)

                           min_delta=0, 
                           patience=10, 
                           verbose=1)

# Reduces learning rate when monitor value plateaus

                              factor=0.316,
