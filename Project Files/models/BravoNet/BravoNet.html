# Saves the best model based on monitor value between epochs

                             monitor="val_accuracy", 
                             verbose=1, 
                             save_best_only=True, 
                             period=1)

# Stops training early when no improvement for monitor value

                           min_delta=0, 
                           patience=15, 
                           verbose=1)

# Reduces learning rate when monitor value plateaus

                              factor=0.1,

                              verbose=1)
